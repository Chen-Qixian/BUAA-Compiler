int n;
int a[10],flag[10];

void perm(int k){
	int i;
	if(k==n){
		i = 0;		
		while(i < n){
			printf(a[i]);
			i = i + 1;
		}
		return;
	}
	i = 1;
	while(i <= n){
		if(flag[i] == 0){
			flag[i] = 1;
			a[k]=i;
			perm(k+1);
			flag[i] = 0;
		}
		i = i + 1;
	}
}
void main(){
	scanf(n);
	perm(0);
}
